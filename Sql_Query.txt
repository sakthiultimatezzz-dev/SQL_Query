SELECT 
    Customers.CustomerID, 
    Customers.CustomerName,
    SUM(OrderDetails.Quantity) AS TotalQuantity -- Added an aggregation for the GROUP BY to make sense
FROM Customers
INNER JOIN Orders 
    ON Customers.CustomerID = Orders.CustomerID
INNER JOIN OrderDetails 
    ON Orders.OrderID = OrderDetails.OrderID
WHERE 
    Orders.OrderDate BETWEEN '1996-07-04' AND '1997-02-12'
    AND Customers.City IS NOT NULL 
    AND Orders.CustomerID IN (1, 2, 3, 4, 5,6,7,8)
GROUP BY 
    Customers.CustomerID, 
    Customers.CustomerName;

--------------The "Contribution" Analysis

SELECT 
    parent_sku_id,
    sku_id AS Child_ID,
    SUM(quantity_sold) OVER(PARTITION BY parent_sku_id) AS Parent_Total_Qty,
    quantity_sold AS Child_Qty,
    ROUND(100.0 * quantity_sold / SUM(quantity_sold) OVER(PARTITION BY parent_sku_id), 2) AS Percent_Contribution
FROM Sales_Table;

---------------SQL Aggregation

SELECT 
    p.sku_id AS Parent_ID,
    p.product_name AS Product_Line,
    SUM(s.quantity_sold) AS Total_Units,
    SUM(s.total_price) AS Total_Revenue
FROM Products p
JOIN Products c ON p.sku_id = c.parent_sku_id -- Joins Parent to its Children
JOIN Sales s ON c.sku_id = s.sku_id            -- Joins Children to Sales data
GROUP BY p.sku_id, p.product_name
ORDER BY Total_Revenue DESC;

CREATE TRIGGER check_price_before_insert
BEFORE INSERT ON Products
FOR EACH ROW
BEGIN
    IF NEW.Price < 0 THEN
        SET NEW.Price = 0;
    END IF;
END;

-----------------After

CREATE TRIGGER log_after_delete
AFTER DELETE ON Products
FOR EACH ROW
BEGIN
        INSERT INTO Product_Audit_Log (ProductID, Action, LogDate)
    VALUES (OLD.ProductID, 'DELETED', NOW());
END;

CREATE TRIGGER prevent_hard_delete
BEFORE DELETE ON Employees
FOR EACH ROW
BEGIN
    -- 1. Update the record to mark it as inactive instead
    UPDATE Employees 
    SET is_active = 0 
    WHERE EmployeeID = OLD.EmployeeID;

    -- 2. Raise an error to 'cancel' the original DELETE command
    SIGNAL SQLSTATE '45000' 
    SET MESSAGE_TEXT = 'Hard delete prevented. Record marked as inactive.';
END;

DELIMITER //

CREATE PROCEDURE GiveRaise(
    IN emp_id INT, 
    IN raise_amount DECIMAL(10,2)
)
BEGIN
    -- Step 1: Update the salary
    UPDATE Employees 
    SET Salary = Salary + raise_amount 
    WHERE EmployeeID = emp_id;
    
    -- Step 2: Select the new details to confirm
    SELECT EmployeeName, Salary AS NewSalary 
    FROM Employees 
    WHERE EmployeeID = emp_id;
END //

DELIMITER ;
